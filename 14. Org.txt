SumOfNumbers.java 
package dowhile; 
public class SumOfNumbers { 
 public static void main(String[] args) { 
  int i = 1, sum = 0; 
  do { 
  sum += i; //sum=sum+i; 
  i++; 
  } while (i <= 10); 
  System.out.println("The sum of numbers from 1 to 10 is: " + sum); 
  } 
  public static int calculateSum(int start, int end) { 
  int i = start, sum = 0; 
  do {  
  sum += i; 
  i++; 
  } while (i <= end); 
    return sum; 
  } 
 }
